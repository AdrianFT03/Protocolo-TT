\begin{UseCase}{CUH1}{Gestionar Horario}{

	Permite realizar las operaciones necesarias para mantener el horario del actualizado conforme a su inscripción actual.\\

	El actor podrá tener la gestión de sus \cdtRef{gls:unidadDeAprendizaje}{Unidades de Aprendizaje} conforme a su inscripción a un \cdtRef{gls:cicloEscolar}{Ciclo escolar}.

\begin{UCtrayectoria}

    \UCpaso[\UCactor] Solicita gestionar su horario presionando la opción \cdtButton{Horarios} del menú.
    
    \UCpaso[\UCsist] Obtiene el nombre, día, horario y grupo de las Unidades de Aprendizaje registradas.
        
    \UCpaso[\UCsist] Muestra la pantalla \cdtIdRef{IUH1}{Horarios} con la información obtenida.
        
    \UCpaso[\UCactor] \label{CUH1:Gestiona} Gestiona su horario con el botón \btnEditar.
        
\end{UCtrayectoria}
